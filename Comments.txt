            string cardNum = txtCreditCardNo.Text;

            int day = Convert.ToInt32(datevalue.Day.ToString());
            int month = Convert.ToInt32(datevalue.Month.ToString());
            int year = Convert.ToInt32(datevalue.Year.ToString());

            string monthCompare = DateTime.Now.Month.ToString();
            string yearCompare = DateTime.Now.Year.ToString();

            if (cardNum.Length == 16 && dateMonth >= month && dateYear >= year && cvvNum.Length == 3)
            {
                MessageBox.Show("You are now a Subscribed user, Thank You for using E-Library System.");
                
            }
            else
            {
                MessageBox.Show("Sorry Your Credit Card Details were incorrect you are now Registered as a Non-Subscribed user, Thank You. ");

            }

            if(userDetailsObj.First_Name != "([A-Z][a-z])" )
            {
                validUD = false;
                objSB.Append(Environment.NewLine + "First Name Should start with Capital letter");
            }

            if (userDetailsObj.Last_Name != "([A-Z][a-z])")
            {
                validUD = false;
                objSB.Append(Environment.NewLine + "Last Name Should start with Capital letter");
            }

            if (dobValid < 18)
            {
                validUD = false;
                objSB.Append(Environment.NewLine + "Age should be 18");
            }
            if(userDetailsObj.Land_Line_Number != "([0-9]{3}-[0-9]{4}-[0-9]{4})")
            {
                validUD = false;
                objSB.Append(Environment.NewLine + "Landline number should be in xxx-xxxx-xxxx format");

            }
            if (userDetailsObj.Mobile_Number != "([0-9]{3}-[0-9]{4}-[0-9]{4})")
            {
                validUD = false;
                objSB.Append(Environment.NewLine + "Mobile number should be in xxx-xxxx-xxxx format");

            }
            if (!(userDetailsObj.User_Type.Equals("Subscriber") || userDetailsObj.User_Type.Equals("Non Subscriber") || userDetailsObj.User_Type.Equals("Administrator")))
            {
                validUD = false;
                objSB.Append(Environment.NewLine + "User Type should be Subscriber or Nonsubscriber or Administrator.");
            }
            if (!(userDetailsObj.Gender.Equals("M") || userDetailsObj.Gender.Equals("F") || userDetailsObj.Gender.Equals("O")))
            {
                validUD = false;
                objSB.Append(Environment.NewLine + "User Type should be Subscriber or Nonsubscriber or Administrator.");
            }
            if (validUD == false)
                 throw new ELibraryMgmtSysException(objSB.ToString());
                //throw new Exception(objSB.ToString());

            return validUD;
        }

        //add user details
        public bool AddUserDetails(User_Details objUser)
        {
            
            bool AddUD = false;
            //try catch is used here  to handle exception.
            try
            {
                if (ValidateUser(objUser))
                {
                    UDdbContext.User_Details.Add(objUser);
                    int i = UDdbContext.SaveChanges();
                    if (i > 0)
                    {
                        AddUD = true;
                    }

                }
            }
            catch (ELibraryMgmtSysException ue)
            {
                throw ue;
            }
            catch (Exception ex)
            {
                throw ex;
            }
            return AddUD;
        }

        //login user
        public bool Login(string userID, string password, string usertype)
        {
            List<User_Details> userList = UDdbContext.User_Details.ToList();
            bool loginUser = false;
            try
            {
                foreach (var user in userList)
                {
                    if (user.Password == password && user.User_ID == userID && user.User_Type == usertype)
                        loginUser = true;
                }
            }
            catch (ELibraryMgmtSysException)
            {
                throw;
            }
            catch (Exception ex)
            {
                throw ex;
            }
            return loginUser;
        }
